# SF_EscribirRegistrosFuenteNovedadesNominaExcel
# Escribe el resultado de gListaRegistrosFuenteNovedadesNomina en la hoja registrosFuenteNovedadesNomina del archivo de analisis de la corrida
SET gEstadoEscrituraRegistrosFuenteNovedadesNominaExcel TO $'''INICIANDO'''
SET gMensajeError TO $''''''
SET gEsErrorControlado TO False
SET gTotalFilasRegistrosFuenteNovedadesNominaEscritas TO 0
SET gTotalFilasRegistrosFuenteNovedadesNominaConErrorFormato TO 0
SET gHayFilasRegistrosFuenteNovedadesNominaEscritas TO False
SET gExcelRegistrosFuenteNovedadesNominaAbierto TO False
SET gNombreHojaRegistrosFuenteNovedadesNomina TO $'''registrosFuenteNovedadesNomina'''
BLOCK 
ON BLOCK ERROR
    SET gEstadoEscrituraRegistrosFuenteNovedadesNominaExcel TO $'''ERROR'''
    SET gEsErrorControlado TO True
    THROW ERROR
END
    File.WriteText File: gArchivoLog TextToWrite: $'''__________________________________________________
Inicio escritura registros fuente novedades nomina Excel | Archivo: %gRutaArchivoAnalisisNovedades% | Registros: %gListaRegistrosFuenteNovedadesNomina.Count%''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
    IF gListaRegistrosFuenteNovedadesNomina.Count = 0 THEN
        SET gEstadoEscrituraRegistrosFuenteNovedadesNominaExcel TO $'''SIN_DATOS'''
        SET gMensajeError TO $'''No hay registros fuente de novedades nomina para escribir en Excel'''
        File.WriteText File: gArchivoLog TextToWrite: $'''SIN_DATOS | %gMensajeError%''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
        EXIT FUNCTION
    END
    IF (File.IfFile.DoesNotExist File: gRutaArchivoAnalisisNovedades) THEN
        SET gEstadoEscrituraRegistrosFuenteNovedadesNominaExcel TO $'''ERROR'''
        SET gMensajeError TO $'''No existe el archivo Excel de analisis de novedades para escritura de registros fuente'''
        File.WriteText File: gArchivoLog TextToWrite: $'''ERROR | %gMensajeError% | Ruta: %gRutaArchivoAnalisisNovedades%''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
        EXIT FUNCTION
    END
    Excel.LaunchExcel.LaunchAndOpenUnderExistingProcess Path: gRutaArchivoAnalisisNovedades Visible: False ReadOnly: False Instance=> ExcelInstance
    SET gExcelRegistrosFuenteNovedadesNominaAbierto TO True
    SET gExisteHojaRegistrosFuenteNovedadesNomina TO False
    SET gExisteSheet1RegistrosFuenteNovedadesNomina TO False
    Excel.GetAllWorksheets Instance: ExcelInstance Worksheets=> gHojasArchivoRegistrosFuenteNovedadesNomina
    LOOP FOREACH CurrentSheetRegistrosFuenteNovedadesNomina IN gHojasArchivoRegistrosFuenteNovedadesNomina
        IF CurrentSheetRegistrosFuenteNovedadesNomina = gNombreHojaRegistrosFuenteNovedadesNomina THEN
            SET gExisteHojaRegistrosFuenteNovedadesNomina TO True
        ELSE IF CurrentSheetRegistrosFuenteNovedadesNomina = $'''Sheet1''' THEN
            SET gExisteSheet1RegistrosFuenteNovedadesNomina TO True
        END
    END
    IF gExisteHojaRegistrosFuenteNovedadesNomina = False THEN
        IF gExisteSheet1RegistrosFuenteNovedadesNomina = True THEN
            Excel.CopyExcelWorksheet.CopyWorksheetWithName SourceInstance: ExcelInstance SourceWorksheetName: $'''Sheet1''' TargetInstance: ExcelInstance TargetWorksheetName: gNombreHojaRegistrosFuenteNovedadesNomina PasteWorksheetPosition: Excel.WorksheetPosition.First
            File.WriteText File: gArchivoLog TextToWrite: $'''OK | Hoja %gNombreHojaRegistrosFuenteNovedadesNomina% creada a partir de Sheet1''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
        ELSE
            SET gEstadoEscrituraRegistrosFuenteNovedadesNominaExcel TO $'''ERROR'''
            SET gMensajeError TO $'''No existe la hoja registrosFuenteNovedadesNomina ni una hoja Sheet1 para reutilizar en el archivo de analisis'''
            File.WriteText File: gArchivoLog TextToWrite: $'''ERROR | %gMensajeError% | Archivo: %gRutaArchivoAnalisisNovedades%''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
            Excel.CloseExcel.CloseAndSave Instance: ExcelInstance
            SET gExcelRegistrosFuenteNovedadesNominaAbierto TO False
            EXIT FUNCTION
        END
    END
    Excel.SetActiveWorksheet.ActivateWorksheetByName Instance: ExcelInstance Name: gNombreHojaRegistrosFuenteNovedadesNomina
    Excel.ReadFromExcel.ReadCell Instance: ExcelInstance StartColumn: $'''A''' StartRow: 1 GetCellContentsMode: Excel.GetCellContentsMode.PlainText CellValue=> gCabeceraRegistrosFuenteNovedadesNominaA1
    IF IsEmpty(gCabeceraRegistrosFuenteNovedadesNominaA1) THEN
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''IdEjecucion''' Column: $'''A''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''ArchivoOrigen''' Column: $'''B''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''HojaOrigen''' Column: $'''C''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''FilaExcel''' Column: $'''D''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''Ciudad''' Column: $'''E''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''NombreEmpleado''' Column: $'''F''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''CedulaFuente''' Column: $'''G''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''Area''' Column: $'''H''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''SueldoBaseTexto''' Column: $'''I''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''AuxExtralegalBaseTexto''' Column: $'''J''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''ConceptoFuenteExcel''' Column: $'''K''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''ColumnaOrigenExcel''' Column: $'''L''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''ValorConceptoFuente''' Column: $'''M''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''NovedadTextoOriginal''' Column: $'''N''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''NovedadAsociadaConcepto''' Column: $'''O''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''MetodoAsociacionNovedad''' Column: $'''P''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''ConfianzaAsociacion''' Column: $'''Q''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''RequiereRevision''' Column: $'''R''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''EstadoRevisionUsuario''' Column: $'''S''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''DecisionUsuario''' Column: $'''T''' Row: 1
        Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''ObservacionRevisionUsuario''' Column: $'''U''' Row: 1
    END
    SET gFilaEscrituraRegistrosFuenteNovedadesNominaActual TO 2
    SET gBuscarFilaLibreRegistrosFuenteNovedadesNomina TO True
    LOOP WHILE gBuscarFilaLibreRegistrosFuenteNovedadesNomina = True
        Excel.ReadFromExcel.ReadCell Instance: ExcelInstance StartColumn: $'''A''' StartRow: gFilaEscrituraRegistrosFuenteNovedadesNominaActual GetCellContentsMode: Excel.GetCellContentsMode.PlainText CellValue=> gValorControlFilaRegistrosFuenteNovedadesNomina
        IF IsEmpty(gValorControlFilaRegistrosFuenteNovedadesNomina) THEN
            SET gBuscarFilaLibreRegistrosFuenteNovedadesNomina TO False
        ELSE
            Variables.IncreaseVariable Value: gFilaEscrituraRegistrosFuenteNovedadesNominaActual IncrementValue: 1
        END
    END
    LOOP FOREACH CurrentItem IN gListaRegistrosFuenteNovedadesNomina
        Text.SplitText.SplitWithDelimiter Text: CurrentItem CustomDelimiter: $'''|''' IsRegEx: False Result=> gPartesRegistrosFuenteNovedadesNominaExcelActual
        IF gPartesRegistrosFuenteNovedadesNominaExcelActual.Count < 18 THEN
            Variables.IncreaseVariable Value: gTotalFilasRegistrosFuenteNovedadesNominaConErrorFormato IncrementValue: 1
            File.WriteText File: gArchivoLog TextToWrite: $'''ERROR | Registro fuente novedades nomina con formato invalido | Registro: %CurrentItem%''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
        ELSE
            SET gArchivoOrigenRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[0]
            SET gHojaOrigenRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[1]
            SET gFilaExcelRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[3]
            SET gCiudadRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[4]
            SET gNombreEmpleadoRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[5]
            SET gCedulaFuenteRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[6]
            SET gAreaRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[7]
            SET gSueldoBaseRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[8]
            SET gAuxExtralegalRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[9]
            SET gConceptoFuenteRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[10]
            SET gColumnaOrigenRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[11]
            SET gValorConceptoRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[12]
            SET gNovedadTextoOriginalRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[13]
            SET gNovedadAsociadaRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[14]
            SET gMetodoAsociacionRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[15]
            SET gConfianzaAsociacionRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[16]
            SET gRequiereRevisionRegistrosFuenteNovedadesNominaExcelActual TO gPartesRegistrosFuenteNovedadesNominaExcelActual[17]
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gIdEjecucion Column: $'''A''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gArchivoOrigenRegistrosFuenteNovedadesNominaExcelActual Column: $'''B''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gHojaOrigenRegistrosFuenteNovedadesNominaExcelActual Column: $'''C''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gFilaExcelRegistrosFuenteNovedadesNominaExcelActual Column: $'''D''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gCiudadRegistrosFuenteNovedadesNominaExcelActual Column: $'''E''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gNombreEmpleadoRegistrosFuenteNovedadesNominaExcelActual Column: $'''F''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gCedulaFuenteRegistrosFuenteNovedadesNominaExcelActual Column: $'''G''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gAreaRegistrosFuenteNovedadesNominaExcelActual Column: $'''H''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gSueldoBaseRegistrosFuenteNovedadesNominaExcelActual Column: $'''I''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gAuxExtralegalRegistrosFuenteNovedadesNominaExcelActual Column: $'''J''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gConceptoFuenteRegistrosFuenteNovedadesNominaExcelActual Column: $'''K''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gColumnaOrigenRegistrosFuenteNovedadesNominaExcelActual Column: $'''L''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gValorConceptoRegistrosFuenteNovedadesNominaExcelActual Column: $'''M''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gNovedadTextoOriginalRegistrosFuenteNovedadesNominaExcelActual Column: $'''N''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gNovedadAsociadaRegistrosFuenteNovedadesNominaExcelActual Column: $'''O''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gMetodoAsociacionRegistrosFuenteNovedadesNominaExcelActual Column: $'''P''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gConfianzaAsociacionRegistrosFuenteNovedadesNominaExcelActual Column: $'''Q''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: gRequiereRevisionRegistrosFuenteNovedadesNominaExcelActual Column: $'''R''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''Pendiente''' Column: $'''S''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''''' Column: $'''T''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: $'''''' Column: $'''U''' Row: gFilaEscrituraRegistrosFuenteNovedadesNominaActual
            Variables.IncreaseVariable Value: gTotalFilasRegistrosFuenteNovedadesNominaEscritas IncrementValue: 1
            Variables.IncreaseVariable Value: gFilaEscrituraRegistrosFuenteNovedadesNominaActual IncrementValue: 1
        END
    END
    Excel.CloseExcel.CloseAndSave Instance: ExcelInstance
    SET gExcelRegistrosFuenteNovedadesNominaAbierto TO False
    IF gTotalFilasRegistrosFuenteNovedadesNominaEscritas = 0 THEN
        SET gEstadoEscrituraRegistrosFuenteNovedadesNominaExcel TO $'''SIN_DATOS'''
        SET gHayFilasRegistrosFuenteNovedadesNominaEscritas TO False
        File.WriteText File: gArchivoLog TextToWrite: $'''Escritura registros fuente novedades nomina Excel finalizada sin filas escritas''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
    ELSE
        SET gEstadoEscrituraRegistrosFuenteNovedadesNominaExcel TO $'''OK'''
        SET gHayFilasRegistrosFuenteNovedadesNominaEscritas TO True
        File.WriteText File: gArchivoLog TextToWrite: $'''Resumen escritura registros fuente novedades nomina Excel | Filas escritas: %gTotalFilasRegistrosFuenteNovedadesNominaEscritas% | Filas con error formato: %gTotalFilasRegistrosFuenteNovedadesNominaConErrorFormato% | Archivo: %gRutaArchivoAnalisisNovedades%''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
    END
END
IF $'''gEsErrorControlado''' = True THEN
    ERROR => gMensajeError
    IF gExcelRegistrosFuenteNovedadesNominaAbierto = True THEN
        Excel.CloseExcel.CloseAndSave Instance: ExcelInstance
        SET gExcelRegistrosFuenteNovedadesNominaAbierto TO False
    END
    File.WriteText File: gArchivoLog TextToWrite: $'''ERROR CONTROLADO | %gMensajeError%''' AppendNewLine: True IfFileExists: File.IfFileExists.Append Encoding: File.FileEncoding.UTF8
    SET gEsErrorControlado TO False
END
