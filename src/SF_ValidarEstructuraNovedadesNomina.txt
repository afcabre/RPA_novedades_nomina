function Normalizar-Texto([string]$valor) {
      if ([string]::IsNullOrWhiteSpace($valor)) {
          return \"\"
      }

      $textoFormD = $valor.Normalize([Text.NormalizationForm]::FormD)
      $sb = New-Object System.Text.StringBuilder

      foreach ($char in $textoFormD.ToCharArray()) {
          $unicodeCategory = [Globalization.CharUnicodeInfo]::GetUnicodeCategory($char)
          if ($unicodeCategory -ne [Globalization.UnicodeCategory]::NonSpacingMark) {
              [void]$sb.Append($char)
          }
      }

     return $sb.ToString().ToLowerInvariant().Trim()
  }

function Obtener-Columna-Novedades([string]$textoCabecera) {
      $columnas = $textoCabecera -split \',\'

      for ($i = 0; $i -lt $columnas.Count; $i++) {
          $valorNormalizado = (Normalizar-Texto $columnas[$i]).Trim()
          if ($valorNormalizado -eq \"novedades\") {
              return ([char](65 + $i)).ToString()
          }
      }

      return \"\"
  }

function Es-Cabecera-Nomina([string]$texto) {
    $contenidoNormalizado = Normalizar-Texto $texto

    $requiredFixedHeaders = @(
        \"ciudad\",
        \"nombre\",
        \"cedula de ciudadania\",
        \"area\",
        \"sueldo\",
        \"aux. extralegal\",
        \"novedades\"
    )

    $optionalHeaders = @(
        \"aux. traslado ciudad/ aux. rodamiento\",
        \"horas operador\",
        \"bonos operador no prestacional\",
        \"hospedaje\",
        \"gastos representacion\",
        \"transporte\",
        \"viaticos no salariales\",
        \"descuentos\"
    )

    $requiredFixedMatches = 0
    foreach ($requiredHeader in $requiredFixedHeaders) {
        if ($contenidoNormalizado.Contains($requiredHeader)) {
            $requiredFixedMatches++
        }
    }

    $optionalMatches = 0
    foreach ($optionalHeader in $optionalHeaders) {
        if ($contenidoNormalizado.Contains($optionalHeader)) {
            $optionalMatches++
        }
    }

    if ($requiredFixedMatches -eq $requiredFixedHeaders.Count -and $optionalMatches -ge 1) {
        return $true
    }

    return $false
}
